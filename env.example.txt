# =============================================================================

# COMO USAR:
#
#   • Python (desenvolvimento): copie este arquivo para .env na RAIZ do projeto
#     (mesma pasta que main.py). Ex.: cp env.example.txt .env
#
#   • .exe (cliente): copie para .env na MESMA PASTA do Gerente.exe.
#     O Gerente procura .env ao lado do executável.
#
# O .env é opcional: sem ele, usam-se os padrões abaixo. No .exe, se FLASK_HOST
# não estiver definido, o padrão é 0.0.0.0 (LAN).
#
# =============================================================================

# --- Servidor -----------------------------------------------------------------

# Porta HTTP
PORT=5000

# Segunda porta (apenas com --duplo ou DUPLO_SERVIDOR=1)
PORT2=5001

# Host de escuta:
#   - 127.0.0.1: só esta máquina (mais seguro)
#   - 0.0.0.0:   aceita outros PCs na rede (necessário para "Descobrir Servidores")
# No .exe, se omitido, o padrão é 0.0.0.0.
FLASK_HOST=0.0.0.0

# Modo debug (true/false). Em false, o .exe é gerado sem janela de terminal.
FLASK_DEBUG=false

# Threads do Waitress (servidor HTTP)
WAITRESS_THREADS=8

# --- Modos de execução -------------------------------------------------------

# 1= silencioso, 0= normal
SILENT_MODE=0

# 1= usar ícone na bandeja, 0= não
USE_TRAY=1

# 1= iniciar dois servidores (portas PORT e PORT2), 0= um só
DUPLO_SERVIDOR=0

# --- Log de debug ------------------------------------------------------------

# Caminho do arquivo de log (apenas em modo debug)
DEBUG_LOG_PATH=.debug/debug.log

# --- Banco de dados ----------------------------------------------------------

# Caminho do SQLite. Se omitido: data/pacientes.db (relativo à pasta do .exe ou do projeto)
# DB_PATH=data/pacientes.db

# Exemplos para rede:
#   Windows: DB_PATH=\\SERVIDOR\pasta\pacientes.db  ou  DB_PATH=Z:\pasta\pacientes.db
#   Linux:   DB_PATH=/mnt/rede/pacientes.db

# --- Sincronização / Descobrir servidores ------------------------------------

# Descoberta na LAN:
#   zeroconf (padrão): mDNS, zero config, sem scan. Tipo: _gerentepaciente._http._tcp.local
#   scan: varredura /24, SYNC_TARGETS ou SYNC_SCAN_CIDRS; líder = menor IP, /register
# DISCOVERY=zeroconf

# Tipo de serviço mDNS (só com DISCOVERY=zeroconf). Padrão: _gerentepaciente._http._tcp.local.
# ZEROCONF_SERVICE_TYPE=_gerentepaciente._http._tcp.local.

# --- Só com DISCOVERY=scan ---------------------------------------------------

# IPs ou hosts fixos para varrer. Ex.: 192.168.1.10;192.168.1.20
# SYNC_TARGETS=

# Redes em notação CIDR. Ex.: 192.168.1.0/24
# SYNC_SCAN_CIDRS=

# Limite de alvos na varredura
# SYNC_MAX_TARGETS=1024

# Intervalo em segundos entre varreduras. Padrão: 15
# LEADER_SCAN_INTERVAL=15
