# Configurações do Sistema de Gestão de Pacientes
# Copie este arquivo para .env e ajuste as configurações conforme necessário

# Porta do servidor Flask
PORT=5000

# Host do servidor Flask
FLASK_HOST=127.0.0.1

# Modo debug do Flask (true/false)
FLASK_DEBUG=false

# Modo silencioso - desabilita logs verbosos (1 para ativar, 0 para desativar)
SILENT_MODE=0

# Usar tray icon (1 para ativar, 0 para desativar)
USE_TRAY=1

# Caminho do arquivo de log de debug
DEBUG_LOG_PATH=.cursor/debug.log

# Configurações do banco de dados
# DB_PATH será definido automaticamente, mas pode ser sobrescrito aqui
# 
# Para banco de dados local (padrão):
# DB_PATH=data/pacientes.db
#
# Para banco de dados compartilhado em rede (Windows):
# DB_PATH=\\NOME_DO_SERVIDOR\pasta_compartilhada\pacientes.db
# ou
# DB_PATH=Z:\pasta_compartilhada\pacientes.db  (se mapeado como unidade)
#
# Para banco de dados compartilhado em rede (Linux/Mac):
# DB_PATH=/mnt/rede/pacientes.db
# ou
# DB_PATH=/path/to/network/share/pacientes.db
#
# AVISO: Compartilhar SQLite via rede pode ser lento e propenso a erros.
# Recomenda-se usar sincronização periódica ou migrar para PostgreSQL/MySQL
# em cenários de múltiplos servidores.
